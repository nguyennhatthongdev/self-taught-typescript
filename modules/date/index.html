Method Description
getFullYear() Get the year as a four digit number (yyyy)
getMonth() Get the month as a number (0-11)
getDate() Get the day as a number (1-31)
getHours() Get the hour (0-23)
getMinutes() Get the minute (0-59)
getSeconds() Get the second (0-59)
getMilliseconds() Get the millisecond (0-999)
getTime() Get the time (milliseconds since January 1, 1970)
getDay() Get the weekday as a number (0-6)
Date.now() Get the time. ECMAScript 5.

1. Lấy ngày cuối cùng của tháng

function getLastDayOfMonth(date) {
    return new Date(date.getFullYear(), date.getMonth() + 1, 0);
}

Example:

const date = new Date("2022-09-01");
const lastDayOfMonth = getLastDayOfMonth(date).getDate();
// log
console.log("lastDayOfMonth", lastDayOfMonth); // lastDayOfMonth 30

3. Lấy số ngày khoảng cách giữa 2 ngày

function inDays(dateStart, dateEnd) {
    return (dateEnd.getTime() - dateStart.getTime()) / (24 * 3600 * 1000);
}

Example:

const dateStart = new Date('2022-08-01')
const dateEnd = new Date('2022-09-31')
const inDay = inDays(dateStart, dateEnd);
// log
console.log("inDay", inDay); // inDay 61

4. Lấy số tuần khoảng cách giữa 2 ngày

function inWeeks(dateStart, dateEnd) {
    return (dateEnd.getTime() - dateStart.getTime()) / (24 * 3600 * 1000 * 7);
}

Example:

const dateStart = new Date('2022-08-01')
const dateEnd = new Date('2022-09-31')
const inWeek = inWeeks(dateStart, dateEnd);
// log
console.log("inWeek", inWeek); // inWeek 8.714285714285714

5. Lấy số tháng khoảng cách giữa 2 ngày

function inMonths(dateStart, dateEnd) {
    let dateStartYear = dateStart.getFullYear();
    let dateEndYear = dateEnd.getFullYear();
    let dateStartMonth = dateStart.getMonth();
    let dateEndMonth = dateEnd.getMonth();

    return (dateEndMonth + 12 * dateEndYear) - (dateStartMonth + 12 * dateStartYear);
}

Example:

const dateStart = new Date('2022-08-01')
const dateEnd = new Date('2022-09-31')
const inMonth = inMonths(dateStart, dateEnd);
// log
console.log("inMonth", inMonth); // inMonth 2

11. Lấy ngày làm việc

function getBusinessDatesCount(startDate, endDate) {
    let count = 0;
    const curDate = new Date(startDate.getTime());
    while (curDate <= endDate) {
        const dayOfWeek = curDate.getDay();
        if (dayOfWeek !== 0 && dayOfWeek !== 6) count++;
        curDate.setDate(curDate.getDate() + 1);
    }
    return count;
}

Example:

const dateStart = new Date('05/03/2016');
const dateEnd = new Date('05/10/2016');
const numOfDates = getBusinessDatesCount(dateStart, dateEnd);
// log
console.log("getBusinessDatesCount", numOfDates); // getBusinessDatesCount 6

12. Cộng thêm 1 ngày

function nextDay(date) {
    return new Date(date.getFullYear(), date.getMonth(), date.getDate() + 1);
}

Example:

const date = new Date('2022-08-31');
const next = nextDay(date).toLocaleString();
// log
console.log("nextDay", next); // nextDay 9/1/2022, 12:00:00 AM
